# @ for function define
@ function_name ( param1: type, param2: type ): type1, type2, type3 {
	statement
	statement; statement; statement;
	out return1, return2, return3
}

var r1, r2, r3 = function_name() // return all out
var s1, s2, = function_name() // it will return first to s1 and s2
var t1 = function_name() // it will return the first only

	out math.pow(x, y);

# conditional
if (conditional) statements else else_statements;

if (conditional) 
	statements;
else 
	statements;

# when (switch statement)
when(conditional)  // switch
	value { statements }
	value { statements; statements; }
	value { statements; statements }


when (value) {
    1 { std.out("one") }
    2 { std.out("two") }
    3 { std.out("three") }
    else { std.out("other") }
}

# while loop

	statements;

	statements; statements; 
} (true_condition )
